function greet(name) {
    return `Hello, ${name}!`;
}

// Arrow function (ES6+) — modern standard
const multiply = (a, b) =&gt; a * b;  // Implicit return
const square = n =&gt; n * n;          // Single param, no parentheses needed

// Arrow with block body (explicit return)
const calculateTax = (amount, rate) =&gt; {
    const tax = amount * (rate / 100);
    return amount + tax;
};

Async JavaScript: Promises and Async/Await</h2><p>Asynchronous programming handles operations that take time — network requests, file reads, database queries — without freezing your application.</p><div class="code-box">// Async/await — modern, readable async code
async function fetchUserData(userId) {
    try {
        const response = await fetch(`https://api.example.com/users/${userId}`);
        
        if (!response.ok) {
            throw new Error(`HTTP error! Status: ${response.status}`);
        }
        
        const data = await response.json();
        return data;
    } catch (error) {
        console.error("Failed to fetch user:", error.message);
        throw error; // Re-throw for caller to handle
    }
}

// Run multiple async operations in parallel
async function loadDashboard(userId) {
    const [user, posts, notifications] = await Promise.all([
        fetchUserData(userId),
        fetchPosts(userId),
        fetchNotifications(userId)
    ]);
    return { user, posts, notifications };
}</div><div class="warning-box">Always use try/catch with async/await. Unhandled promise rejections crash Node.js applications and silently fail in browsers.
