function ShoppingCart() {
  const [items, setItems] = useState([]);
  const [total, setTotal] = useState(0);
  
  const addItem = (product) =&gt; {
    setItems(prev =&gt; [...prev, product]);
    setTotal(prev =&gt; prev + product.price);
  };
  
  const removeItem = (id) =&gt; {
    const item = items.find(i =&gt; i.id === id);
    setItems(prev =&gt; prev.filter(i =&gt; i.id !== id));
    setTotal(prev =&gt; prev - item.price);
  };
  
  return (
    &lt;div&gt;
      &lt;h2&gt;Cart ({items.length} items)&lt;/h2&gt;
      {items.map(item =&gt; (
        &lt;div key={item.id}&gt;
          {item.name} — ${item.price}
          &lt;button onClick={() =&gt; removeItem(item.id)}&gt;Remove&lt;/button&gt;
        &lt;/div&gt;
      ))}
      &lt;strong&gt;Total: ${total.toFixed(2)}&lt;/strong&gt;
    &lt;/div&gt;
  );
}</div><h2>5. useEffect: Side Effects</h2><div class="code-box">import { useState, useEffect } from "react";

function UserProfile({ userId }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  
  useEffect(() =&gt; {
    // Runs after render when userId changes
    setLoading(true);
    setError(null);
    
    fetch(`/api/users/${userId}`)
      .then(res =&gt; {
        if (!res.ok) throw new Error("User not found");
        return res.json();
      })
      .then(data =&gt; {
        setUser(data);
        setLoading(false);
      })
      .catch(err =&gt; {
        setError(err.message);
        setLoading(false);
      });
  }, [userId]); // Re-run when userId changes
  
  if (loading) return &lt;div&gt;Loading...&lt;/div&gt;;
  if (error) return &lt;div&gt;Error: {error}&lt;/div&gt;;

const AuthContext = createContext();

// Wrap your entire app with this provider
export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  
  const login = async (email, password) =&gt; {
    const res = await fetch("/api/auth/login", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ email, password })
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error);
    setUser(data.user);
    localStorage.setItem("token", data.token);
  };
  
  const logout = () =&gt; {
    setUser(null);
    localStorage.removeItem("token");
  };
  
  return (
    &lt;AuthContext.Provider value={{ user, login, logout, isAuthenticated: !!user }}&gt;
      {children}
    &lt;/AuthContext.Provider&gt;
  );
}
